```bash

$ git clone https://github.com/Ekpo-Emmanuel/blog-project

# Access
$ cd Blog

# Create database called "blog_project" 
# import 'blog_project.sql' into MYSQL server
# Run the project
$ localhost/Blog

# The server will initialize in the <http://localhost/Blog>
```
